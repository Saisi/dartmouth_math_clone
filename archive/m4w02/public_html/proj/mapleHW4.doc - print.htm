Mal:=proc(T0,I0, N,F)

local S,E,In,R,P, M, IM,Val,K,Km,T,Tm,C,Cm,v,u,a,b,p,r,x,y,g,Deltat,List,i,Time;

p:=1/4;

x:=1/2;

y:=1/4;

r:=1/12;

Km:=200;

C:=(K*v)/(v-u);

Cm:=(Km*x)/(x-y);

a:=1/20;

b:=9/(20*2*100);

Val:=I0;

List:=[[0,Val]];

Deltat:=(F-T0)/N;

for i from 1 to N

do

S:=Val[1];

E:=Val[2];

In:=Val[3];

R:=Val[4];

P:=Val[5];

IM:= Val[7];

Val:=

[

abs(evalf(

S+Deltat*(

v*(S+E+In+R+P)*(1-(T/C))

-b*S*IM

-u*S

-v*S*(1/8)*(1-(T/C))

))),

abs(evalf(

E+Deltat*(

b*S*IM

+(b*IM*P)/10

-r*E

-u*E

))),

abs(evalf(

In+Deltat*(

r*E-

(u+a+p)*In

))),

 

abs(evalf(

R+Deltat*(

p*In

-(u+g)*R

))),

 

abs(evalf(

P+Deltat*(

v*S*(1/8)*(1-T/C)

+

g*R

-

u*P

-(b*IM*P)/10

))),

abs(evalf(

M+Deltat*(

x*(M+IM)*(1-(Tm/Cm))

-u*M

-b*M*In

))

),

abs(evalf(

IM+Deltat*(

b*M*In

- 2*u*IM

)))

];

Time:=Time+Deltat;

List:=[op(List),[Time,Val]];

od;

List;

end:

ZoomIn:=proc(V,S,E)

local n,M,i,j,K,k,C,N,l,J,p;

M:=[];

n:=nops(S);

for j from 1 to n

do

K:=[];

for i from 1 to n

do

K:=[op(K),diff(S[j],V[i])];

od;

M:=[op(M),K];

od;

M;

C:={};

for k from 1 to n

do

C:={op(C),V[k]=E[k]};

od;

J:=[];

for p from 1 to n

do

J:=[op(J),subs(C,S[p])];

od;

N:=[];

for l from 1 to n

do

N:=[op(N),subs(C,M[l])];

od;
