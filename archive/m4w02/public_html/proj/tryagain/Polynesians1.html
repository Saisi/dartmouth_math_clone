> JPAK1:=proc(T0,I0,N,T)
local i,List,Deltat,Val,Time;
Digits:=5;
Val:=I0;
Time:=T0;
List:=[[0,Val]];
Deltat:=(T-T0)/N;
for i from 1 to N
do
Val:=
[
evalf(
Val[1]+Deltat*(
(2.5/7-1/60)*Val[1]*(1-Val[1]/500)
-
(
2*(33/40)*
(1/(500*2))
)*
Val[2]*Val[1]
-.98*(99)*(1/(500*190))*
Val[3]*Val[1]
)
),
evalf(
Val[2]+Deltat*(
(4/12-1/20)*(1-Val[2]/2)
*Val[2]
)
),
evalf(Val[3]+Deltat*(((1/15-1/26)*Val[3]*(1-Val[3]/190)
-((16/18)*Val[2])*(1/(2*190))
*Val[3])/2
+
(
(16/30)*
(1/(190*500))
)*
Val[1]*Val[3]
)
)
];
Time:=Time+Deltat;
List:=[op(List),[Time,Val]];
od;
List;
end:
