> DEL:=proc(M,n,N,T)
local i,j,List,Deltat,Val,Time,PL;
with(plots);
Digits:=5;
PL:=[];
for j from 1 to n
do
Val:=[cos(j*((2*Pi)/n)),sin(j*((2*Pi)/n))];
Time:=0;
List:=[[0,Val]];
Deltat:=(T-0)/N;
for i from 1 to N
do
Val:=
[
evalf(
Val[1]+Deltat*(M[1][1]*Val[1]+M[1][2]*Val[2]
)
),
evalf(
Val[2]+Deltat*(
M[2][1]*Val[1]+M[2][2]*Val[2]
)
)
];
Time:=Time+Deltat;
List:=[op(List),[Time,Val]];
od;
PL:=[op(PL),listplot(Make3(List),color=red,thickness=2)];
od;
display(PL);
end:

Make3:=proc(List)
local i,NewList;
NewList:=[];
for i from 1 to nops(List)
do
NewList:=[op(NewList),[List[i][2][1],List[i][2][2]]];
od;
NewList
end:

> DEL([[-2,0],[0,-1]],10,250,5);
