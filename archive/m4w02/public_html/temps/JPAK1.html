> JPAK:=proc(T0,I0,N,T)
local i,List,Deltat,Val,Time;
Digits:=5;
Val:=I0;
Time:=T0;
List:=[[0,Val]];
Deltat:=(T-T0)/N;
for i from 1 to N
do
Val:=
[
evalf(
Val[1]+Deltat*(

(1/24-1/36)*Val[1]*(1-Val[1]/250)
-
((10/12)*(1/(250*10)))*
Val[2]*Val[1]
-
((15/12)*(1/(15*250)))*
Val[3]*Val[1]
)
),
evalf(
Val[2]+Deltat*(-(2/12)*Val[2]
+
(2*2*(2/12)*(1/(10*250)))
*Val[1]*Val[2]
-
((1/12)*(1/(10*15)))
*Val[3]*Val[2]
)
)
,
evalf(
Val[3]+Deltat*(

(1/20-1/30)*Val[3]*(1-Val[3]/15)
+
(1/(9*15*250))*
Val[3]*Val[1]
)
)
];
Time:=Time+Deltat;
List:=[op(List),[Time,Val]];
od;
List;
end:


Make1:=proc(List)
local i,NewList;
NewList:=[];
for i from 1 to nops(List)
do
NewList:=[op(NewList),[List[i][1],List[i][2][1]]];
od;
NewList
end:

Make2:=proc(List)
local i,NewList;
NewList:=[];
for i from 1 to nops(List)
do
NewList:=[op(NewList),[List[i][1],List[i][2][2]]];
od;
NewList
end:

Make3:=proc(List)
local i,NewList;
NewList:=[];
for i from 1 to nops(List)
do
NewList:=[op(NewList),[List[i][1],List[i][2][3]]];
od;
NewList
end:
