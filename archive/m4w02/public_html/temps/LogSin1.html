DE2:=proc(T0,I0,N,T)
local i,List,Deltat,Position,Time;
Digits:=5;
Position:=I0;
Time:=T0;
List:=[[Time,Position]];
Deltat:=(T-T0)/N;
for i from 1 to N
do
Position:=evalf(Position+Deltat*
((1/24)*(sin((2*Pi/12)*Time)) + (1/24)-(1/36))*
(Position*(1-(Position/250))));
Time:=evalf(Time+Deltat);
List:=[op(List),[Time,Position]];
od;
List;
end:
