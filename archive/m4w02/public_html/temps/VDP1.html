with(plots);
Digits:=5;
n:=nops(L);
PL:=[];
for j from 1 to n
do
Val:=L[j];
Time:=0;
List:=[[0,Val]];
Deltat:=(T-0)/N;
for i from 1 to N
do
Val:=
[
evalf(
Val[1]+Deltat*(Val[2]
)
),
evalf(
Val[2]+Deltat*(
-Val[1]+Val[2]*(1-(Val[2])^2)
)
)
];
Time:=Time+Deltat;
List:=[op(List),[Time,Val]];
od;
PL:=[op(PL),Make3(List)];
od;
PL;
end:

Make3:=proc(List)
local i,NewList;
NewList:=[];
for i from 1 to nops(List)
do
NewList:=[op(NewList),[List[i][2][1],List[i][2][2]]];
od;
NewList
end:
