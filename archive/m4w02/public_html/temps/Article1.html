> Article:=proc(T0,N,F,w,m,n,k,L,lr)
local S,E,In,R,Val,K,T,v,s,u,a,b,p,C,g,r,Deltat,List,i,Time;
Digits:=5;
Time:=T0;
K:=1.62*w^(-.7);
r:=0.6*w^(-.27);
u:=0.4*w^(-.26);
p:=n*u;
s:=k*u;
g:=r+u;
v:=K*g+u;
C:=(K*v)/(v-u);
a:=(m-1)*u;
b:=L*(2.47)*(10^(-2))*(m+n)*w^(.44);
Val:=[0.7*K,0.3*K,0,0];
List:=[[0,Val]];
Deltat:=(F-T0)/N;
for i from 1 to N
do
S:=Val[1];
E:=Val[2];
In:=Val[3];
R:=Val[4];
T:=Val[1]+Val[2]+Val[4];
Val:=
[
evalf(
S+Deltat*(
v*(1-T/C)*S
-u*S
+lr*R
-b*S*In
)
),
evalf(
E+Deltat*(
b*In*S
-s*E
-u*E
)
)
,
evalf(
In+Deltat*(
s*E
-a*In
-u*In
-p*In
)
),
evalf(
R+Deltat*(
p*In
-u*R
-lr*R
)
)
];
Time:=Time+Deltat;
List:=[op(List),[Time,Val]];
od;
List;
end:
