DE1:=proc(T0,I0,N,T,a)
local i,List,Deltat,Val,Time;
Digits:=10;
Val:=I0;
Time:=T0;
List:=[[0,Val]];
Deltat:=(T-T0)/N;
for i from 1 to N
do
Val:=
[
evalf(
Val[1]+Deltat*(
-(Val[1])^2-a*Val[1]*Val[2]
-(1/a)*Val[1]
-
Val[2]

)
),
evalf(
Val[2]+Deltat*(a*Val[1]*Val[2]+
(1-(1/a))*Val[1]
)
)
];
Time:=Time+Deltat;
List:=[op(List),[Time,Val]];
od;
List;
end:
