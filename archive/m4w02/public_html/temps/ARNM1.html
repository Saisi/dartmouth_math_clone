local i,List,Deltat,Val,Time;
Digits:=5;
Val:=I0;
Time:=T0;
List:=[[0,Val]];
Deltat:=(T-T0)/N;
for i from 1 to N
do
Val:=
[
evalf(
Val[1]+Deltat*(

(1/24-1/36)*Val[1]*(1-Val[1]/250)
-
((10/12)*(1/(250*10)))*
Val[2]*Val[1]
-
((20/12)*(1/(250*30)))*
Val[3]*Val[1]
)
),
evalf(
Val[2]+Deltat*(
(2*(2/12)*(1/250))
*Val[1]*Val[2]-(2/12)*Val[2]
)
)
,
evalf(
Val[3]+Deltat*(

(1/36-1/180)*Val[3]*(1-Val[3]/50)
-
((2/12)*(1/(30*10)))*
Val[3]*Val[2]
+
((5/12)*(1/(30*250)))*
Val[3]*Val[1]

)
)
];
Time:=Time+Deltat;
List:=[op(List),[Time,Val]];
od;
List;
