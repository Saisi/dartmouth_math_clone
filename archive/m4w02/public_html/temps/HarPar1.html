> DEL:=proc(L,N,T,u)
local i,j,List,Deltat,Val,Time,PL,n;
with(plots);
Digits:=5;
n:=nops(L);
PL:=[];
for j from 1 to n
do
Val:=L[j];
Time:=0;
List:=[[0,Val]];
Deltat:=(T-0)/N;
for i from 1 to N
do
Val:=
[
evalf(
Val[1]+Deltat*(Val[2]
)
),
evalf(
Val[2]+Deltat*(
-Val[1]-u*Val[2]*(1)
)
)
];
Time:=Time+Deltat;
List:=[op(List),[Time,Val]];
od;
PL:=[op(PL),Make1(List)];
od;
PL;
end:

Make1:=proc(List)
local i,NewList;
NewList:=[];
for i from 1 to nops(List)
do
NewList:=[op(NewList),[List[i][1],List[i][2][2]]];
od;
NewList
end:
