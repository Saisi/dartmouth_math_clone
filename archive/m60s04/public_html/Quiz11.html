> R:=rand();
randomize(R);

RandomQuiz:=proc(L);
L[stats[random,discreteuniform[1,nops(L)]](1)];
end:


MakeList:=proc(a,b,Chapter)
local i,H;
H:=[];
for i from a to b
do
H:=[op(H),[["Problem",i],["Section",Chapter]]];
od;
H;
end:


> Problems:=[op(MakeList(1,10,"1.2")),op(MakeList(14,16,"1.2"))];

> RandomQuiz(Problems);
