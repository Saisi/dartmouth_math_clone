> with(plots):

CompareS1:=proc(N)
local i,H,L,P,Q,s1,s2;
s1:=0;
s2:=0;
H:=[];
L:=[];
for i from 1 to N
do
s1:= 3^i/(5^i+i+4)+s1;
s2:=3^i/5^i+s2;
H:=[op(H),[i,s1]];
L:=[op(L),[i,s2]];
od;
P:=pointplot({op(H) }, color=red);
Q:=pointplot({op(L) }, color=blue);
display(P,Q);
end:


CompareS1(50);
