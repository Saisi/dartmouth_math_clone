(1) x(t) = t^2-2, y(t) = t^3, z(t) = t^2+1;

(2) x(t) = (6+sin(20 t)) cos(t), y(t) = (6+sin(20 t)) sin(t), z(t) = cos(20 t);

(3) x(t) = t, y(t) = sin(t), z(t) =cos(t);
