   Define a procedure for plotting a function

    and approximating its root using Newton's Method

> newton_plot := proc (f, xstart, n)
local fp, i, x:

print(plot(f(x), x = -1..2)):
fp := diff(f(x), x):

x := xstart:
print (x):
for i from 1 to n do
x := evalf(x - f(x) / fp(x)):
print(x):
od:
end:
