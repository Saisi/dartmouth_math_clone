> "Conservative Forces";

> Explore1:=proc(a,b,c)
local i;
evalf(int(a*diff(a,t)+b*diff(b,t)+c*diff(c,t),t=0..1));
end:

Explore1(t^2,t^3,t);

Explore1(t,t,t);

Explore1(t^3,t^4,t^6);

Explore1(sin(Pi/2*t),(exp(t)-1)/(exp(1)-1),t^6);

Explore1(tan(t*Pi/4), (exp(t^2)-1)/(exp(1)-1),t^6);

> Explore2:=proc(a,b,c)
local i;
evalf(int(a*b*diff(a,t)+b*diff(b,t)+c*diff(c,t),t=0..1));
end:

Explore2(t^2,t^3,t);

Explore2(t,t,t);

Explore2(t^3,t^4,t^6);

Explore2(sin(Pi/2*t),(exp(t)-1)/(exp(1)-1),t^6);

Explore2(tan(t*Pi/4), (exp(t^2)-1)/(exp(1)-1),t^6);
