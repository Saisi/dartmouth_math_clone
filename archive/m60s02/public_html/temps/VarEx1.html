> NormIt:=proc(L,n,M)
local i,H,V1,V2;
H:=[];
for i from 1 to nops(L)
do
H:=[op(H),evalf(L[i]/sqrt(n))];
od;
V1:=Areabargraph(H,-max(-min(op(H)),max(op(H))),max(-min(op(H)),max(op(H))),M);
V2 := plot((1/(sqrt(2*Pi)))*exp(-x^2/(2)),x =-max(-min(op(H)),max(op(H)))..max(-min(op(H)),max(op(H))),color=blue);
display({V1,V2});
end:

> NormIt(D10,10,9);

> NormIt(D100,100,11);

> NormIt(D1000,1000,11);

>
