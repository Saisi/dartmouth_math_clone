> R:=rand();
randomize(R);

RandomQuiz:=proc(L);
L[stats[random,discreteuniform[1,nops(L)]](1)];
end:


MakeList:=proc(a,b,Set,Chapter)
local i,H;
H:=[];
for i from a to b
do
H:=[op(H),[["Problem",i],["Exercise Set",Set],["Chapter",Chapter]]];
od;
H;
end:


> Problems:=MakeList(1,15,A,2);

> RandomQuiz(Problems);
