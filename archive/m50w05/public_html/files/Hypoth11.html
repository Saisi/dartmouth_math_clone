Spikegraph:=proc(distributionlist,xmin,xmax,show,c)
local num,linelist,j,i,templist:
templist:=[]:
if ((distributionlist[1][1]<xmin) or
(distributionlist[nops(distributionlist)][1]>xmax)
) then
lprint(`Note: some data values lie outside the user-defined interval.`)
fi:
for i from 1 to nops(distributionlist) do
if not((distributionlist[i][1]<xmin) or (distributionlist[i][1]>xmax)) then
templist:=[op(templist),
[distributionlist[i][1],
distributionlist[i][2]
]
]:
fi:
od:
num:=nops(templist):
linelist:=[]:
for j from 1 to num do
linelist:=[op(linelist),[templist[j][1],0]]:
linelist:=[op(linelist),[templist[j][1],templist[j][2]]]:
linelist:=[op(linelist),[templist[j][1],0]]:
od:
if show then plot(linelist,style=LINE,color=c) fi;
end:

BinomialPlot:=proc(n,m,p,q)
local length,i,j1,j2,j3,j4,j5,spikelist1,spikelist2,spikelist3,
spikelist4,spikelist5,A1,A2,A3,A4,A5,e:
e:=.1*(1-2/n);
spikelist1:=[]:
spikelist2:=[]:
spikelist3:=[]:
spikelist4:=[]:
spikelist5:=[]:
for j2 from 0 to n
do
spikelist2:=[op(spikelist2),Bin(n,p,j2)] ;
od:

for j1 from 0 to m-1
do
spikelist1:=[op(spikelist1),[j1+e,Bin(n,p,j1)]] ;
od:

for j5 from m to n do
spikelist5:=[op(spikelist5),[j5+e,Bin(n,p,j5)]] ;
od:

for j3 from 0 to m-1
do
spikelist3:=[op(spikelist3),[j3 -e,Bin(n,q,j3)]] ;
od:

for j4 from m to n do
spikelist4:=[op(spikelist4),[j4- e ,Bin(n,q,j4)]] ;
od:

A1:= Spikegraph(spikelist1,0-e,m+e,true,blue);
A5:=Spikegraph(spikelist5,m-1-e,n+e,true,green);

A3:= Spikegraph(spikelist3,0-e,m+1+e,true,yellow);
A4:=Spikegraph(spikelist4,m-1-e,n+e,true,red);

A2:=Spikegraph([[m-.5,(1.1)*max(op(spikelist2))]],m-.5,m-.5,true,black);

display(A1,A5,A3,A4,A2);
end:




Hyp:=proc(n,m,p,q)
local r,s;
r:=sum(
binomial(n,k)*(p^k)*((1-p)^(n-k)),k=m...n
);
s:=sum(
binomial(n,k)*(q^k)*((1-q)^(n-k)),k=0...m-1
);
lprint("There is a",100*r,"percent chance that we accept H1 given  that H0 is correct, and a  ",
100*s,"percent  chance that we accept H0 given  that H1 is correct.  Our cut off was", m-.5);
BinomialPlot(n,m,p,q);
end:
