with(linalg):

Groups :=proc(L,n)
local R,i,k,k1,j,l,H,K;
R := randperm(nops(L));
k1:=nops(L)/n;
k:=floor(k1);
K:=[];
for i from 0 to k-1 do
H:=[i+1];
for j from 1 to n do
H:=[op(H),L[R[n*i+j]]];
od;
K:=[op(K),H];
od;
if nops(L)-n*k > 0
then
H:=[k+1];
for l from n*k+1 to nops(L)
do
H:=[op(H),L[R[l]]];
od;
K:=[op(K),H];
fi;
K;
matrix(K);
end:
